Line Follower Code without PWM

#define LEFT_SENSOR 25
#define RIGHT_SENSOR 26

#define LEFT_LED 18
#define RIGHT_LED 19

void setup() {
  pinMode(LEFT_SENSOR, INPUT_PULLUP);
  pinMode(RIGHT_SENSOR, INPUT_PULLUP);

  pinMode(LEFT_LED, OUTPUT);
  pinMode(RIGHT_LED, OUTPUT);
}

void forward() {
  digitalWrite(LEFT_LED, HIGH);
  digitalWrite(RIGHT_LED, HIGH);
}

void turnLeft() {
  digitalWrite(LEFT_LED, LOW);
  digitalWrite(RIGHT_LED, HIGH);
}

void turnRight() {
  digitalWrite(LEFT_LED, HIGH);
  digitalWrite(RIGHT_LED, LOW);
}

void stopBot() {
  digitalWrite(LEFT_LED, LOW);
  digitalWrite(RIGHT_LED, LOW);
}

void loop() {
  int leftVal = digitalRead(LEFT_SENSOR);
  int rightVal = digitalRead(RIGHT_SENSOR);

  if (leftVal == 1 && rightVal == 1) {
    forward();
  } 
  else if (leftVal == 0 && rightVal == 1) {
    turnRight();
  } 
  else if (leftVal == 1 && rightVal == 0) {
    turnLeft();
  } 
  else {
    stopBot();
  }
}
